function calcNet(salary, deps) {

  salary = Math.max(0, +salary || 0);

  deps   = Math.max(1, +deps   || 1);

  const monthly = salary / 12;


  const pension = Math.round(Math.min(monthly, R.pensionCap) * R.pension);

  const health  = Math.round(monthly * R.health);

  const ltc     = Math.round(health  * R.ltcRatio);

  const employ  = Math.round(monthly * R.employ);


  let ded;

  if      (salary <= 5e6)   ded = salary * 0.70;

  else if (salary <= 15e6)  ded = 3.5e6  + (salary -  5e6) * 0.40;

  else if (salary <= 45e6)  ded = 7.5e6  + (salary - 15e6) * 0.15;

  else if (salary <= 100e6) ded = 12e6   + (salary - 45e6) * 0.05;

  else                      ded = 14.75e6+ (salary -100e6)  * 0.02;

  ded = Math.min(ded, 20e6);


  const taxable = Math.max(0, salary - ded - 1.5e6 * deps);


  let raw;

  if      (taxable <= 14e6)  raw = taxable * 0.06;

  else if (taxable <= 50e6)  raw = taxable * 0.15  - 1.26e6;

  else if (taxable <= 88e6)  raw = taxable * 0.24  - 5.76e6;

  else if (taxable <= 150e6) raw = taxable * 0.35  - 15.44e6;

  else if (taxable <= 300e6) raw = taxable * 0.38  - 19.94e6;

  else if (taxable <= 500e6) raw = taxable * 0.40  - 25.94e6;

  else if (taxable <= 1e9)   raw = taxable * 0.42  - 35.94e6;

  else                       raw = taxable * 0.45  - 65.94e6;

  raw = Math.max(0, raw);


  let cr  = raw <= 1.3e6 ? raw * 0.55 : 715000 + (raw - 1.3e6) * 0.3;

  let cap;

  if      (salary <= 33e6) cap = 740000;

  else if (salary <= 70e6) cap = 740000 - (salary - 33e6) * (80000/37e6);

  else                     cap = Math.max(660000 - (salary - 70e6) * (160000/130e6), 500000);

  cr = Math.min(cr, cap);


  const annIT    = Math.max(0, Math.round(raw - cr));

  const mIT      = Math.round(annIT / 12);

  const mLT      = Math.round(annIT * 0.1 / 12);

  const totalDed = pension + health + ltc + employ + mIT + mLT;

  const grossM   = Math.round(monthly);

  const netM     = grossM - totalDed;

  const eff      = salary > 0 ? (totalDed*12)/salary*100 : 0;


  return { grossM, netM, annualNet: netM*12, annualDed: totalDed*12, eff,

           d: { pension, health, ltc, employ, mIT, mLT, totalDed } };

}
